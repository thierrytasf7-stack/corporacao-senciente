```
import json
import os
from datetime import datetime

def generate_metrics(stories_dir: str) -> dict:
    stories = load_all_stories(stories_dir)
    return {
        'generated_at': datetime.now().isoformat(),
        'total_stories': len(stories),
        'throughput_per_hour': calculate_throughput(stories),
        'avg_latency_per_phase': {
            'TODO_to_EXEC': avg_transition_time(stories, 'TODO', 'EM_EXECUCAO'),
            'EXEC_to_REVISAO': avg_transition_time(stories, 'EM_EXECUCAO', 'PARA_REVISAO'),
        },
        'success_rate': calculate_success_rate(stories),
        'by_agent': count_by_agent(stories),
    }

def save_metrics(metrics: dict, output_path: str) -> None:
    os.makedirs(os.path.dirname(output_path), exist_ok=True)
    with open(output_path, 'w') as f:
        json.dump(metrics, f, indent=2)
```
