## Exemplo de Referencia
```
import zipfile
import os
from datetime import datetime
from glob import glob

def backup_stories(stories_dir: str, backup_dir: str, max_backups: int = 10) -> str:
    os.makedirs(backup_dir, exist_ok=True)
    timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
    zip_path = os.path.join(backup_dir, f'stories_backup_{timestamp}.zip')
    with zipfile.ZipFile(zip_path, 'w', zipfile.ZIP_DEFLATED) as zf:
        for md_file in glob(os.path.join(stories_dir, '*.md')):
            zf.write(md_file, os.path.basename(md_file))
    # Cleanup old backups
    backups = sorted(glob(os.path.join(backup_dir, 'stories_backup_*.zip')))
    for old in backups[:-max_backups]:
        os.remove(old)
    return zip_path
```
